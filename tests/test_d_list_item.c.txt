#include <compiler.h>
#include <list_item.h>

/* Perform tests on a single list item.
 */
void test_single_item(void)
{
	list_item_t list_item;
	
	list_item__ctor(&list_item);
	
	assert(list_item__is_first(&list_item));
	assert(list_item__is_last(&list_item));
	assert(!list_item__in_list(&list_item));
	
	assert(list_item__next(&list_item) == NULL);
	assert(list_item__prev(&list_item) == NULL);
	assert(list_item__list(&list_item) == NULL);
}

void test_append(void)
{
	list_item_t item_a;
	list_item_t item_b;
	
	list_item__ctor(&item_a);
	list_item__ctor(&item_b);
	list_item__append(&item_a, &item_b);
	
	assert(list_item__is_first(&item_a));
	assert(!list_item__is_last(&item_a));
	
	assert(!list_item__is_first(&item_b));
	assert(list_item__is_last(&item_b));
	
	assert(list_item__prev(&item_a) == NULL);
	assert(list_item__next(&item_a) == &item_b);
	assert(list_item__prev(&item_b) == &item_a);
	assert(list_item__next(&item_b) == NULL);
}

void test_prepend(void)
{
	list_item_t item_a;
	list_item_t item_b;
	
	list_item__ctor(&item_a);
	list_item__ctor(&item_b);
	
	list_item__prepend(&item_b, &item_a);
	
	assert(list_item__is_first(&item_a));
	assert(!list_item__is_last(&item_a));
	
	assert(!list_item__is_first(&item_b));
	assert(list_item__is_last(&item_b));
	
	assert(list_item__prev(&item_a) == NULL);
	assert(list_item__next(&item_a) == &item_b);
	assert(list_item__prev(&item_b) == &item_a);
	assert(list_item__next(&item_b) == NULL);
}

void test_delete(void)
{
	list_item_t item_a;
	list_item_t item_b;
	list_item_t item_c;
	
	list_item__ctor(&item_a);
	list_item__ctor(&item_b);
	list_item__ctor(&item_c);
	
	list_item__append(&item_a, &item_b);
	list_item__append(&item_b, &item_c);
	
	list_item__delete(&item_b);
	
	assert(list_item__next(&item_a) == &item_c);
	assert(list_item__prev(&item_c) == &item_a);
	
	assert(list_item__prev(&item_b) == NULL);
	assert(list_item__next(&item_b) == NULL);
}

int main(int argc, char *argv[])
{
	test_single_item();
	test_append();
	test_prepend();
	
	return 0;
}
